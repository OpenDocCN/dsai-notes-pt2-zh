class Discriminator(ndl.nn.Module):
    def __init__(self):
        super().__init__()
        self.net = ndl.nn.Sequential(
            ndl.nn.Linear(2, 20),
            ndl.nn.ReLU(),
            ndl.nn.Linear(20, 10),
            ndl.nn.ReLU(),
            ndl.nn.Linear(10, 2)  # 输出两个类别的分数
        )

    def forward(self, x):
        return self.net(x)

### 生成器更新步骤

生成器的目标是让判别器将其输出误判为真实数据（标签为1）。

![](img/591e45cad1f1d4045687cb5917dafade_148.png)

```python
def update_generator(z, model_G, model_D, loss_fn, opt_G):
    # 生成假数据
    x_fake = model_G(z)
    # 判别器对假数据的预测
    y_fake_pred = model_D(x_fake)
    # 我们希望判别器将假数据判断为“真”（标签1）
    batch_size = z.shape[0]
    ones = ndl.Tensor(np.ones((batch_size,)))
    loss = loss_fn(y_fake_pred, ones)
    # 反向传播并更新生成器参数
    opt_G.zero_grad()
    loss.backward()
    opt_G.step()
    return loss
```

判别器的目标是正确区分真假数据。对于假数据，标签为0；对于真数据，标签为1。

![](img/591e45cad1f1d4045687cb5917dafade_167.png)

![](img/591e45cad1f1d4045687cb5917dafade_169.png)

```python
def update_discriminator(x_real, z, model_G, model_D, loss_fn, opt_D):
    batch_size = x_real.shape[0]
    # 生成假数据，并阻止梯度流向生成器
    with ndl.no_grad():
        x_fake = model_G(z)
    # 判别器对真假数据的预测
    y_fake_pred = model_D(x_fake)
    y_real_pred = model_D(x_real)
    # 创建标签
    zeros = ndl.Tensor(np.zeros((batch_size,)))
    ones = ndl.Tensor(np.ones((batch_size,)))
    # 计算损失：判别器应把假数据判为0，真数据判为1
    loss_fake = loss_fn(y_fake_pred, zeros)
    loss_real = loss_fn(y_real_pred, ones)
    loss = loss_fake + loss_real
    # 反向传播并更新判别器参数
    opt_D.zero_grad()
    loss.backward()
    opt_D.step()
    return loss
```

### 整合训练循环

![](img/591e45cad1f1d4045687cb5917dafade_179.png)

现在，我们将上述步骤整合到一个完整的训练循环中。

![](img/591e45cad1f1d4045687cb5917dafade_181.png)

![](img/591e45cad1f1d4045687cb5917dafade_183.png)

```python
def train_gan(data, model_G, model_D, batch_size=32, epochs=2000):
    opt_G = ndl.optim.SGD(model_G.parameters(), lr=0.01)
    opt_D = ndl.optim.SGD(model_D.parameters(), lr=0.01)
    n = data.shape[0]
    for epoch in range(epochs):
        # 随机选取一批真实数据
        idx = np.random.randint(0, n, batch_size)
        x_real_batch = data[idx]
        # 生成一批随机噪声
        z_batch = ndl.Tensor(np.random.randn(batch_size, 2))
        # 1. 更新判别器
        loss_d = update_discriminator(x_real_batch, z_batch, model_G, model_D, loss_fn, opt_D)
        # 2. 更新生成器
        loss_g = update_generator(z_batch, model_G, model_D, loss_fn, opt_G)
        if epoch % 500 == 0:
            print(f"Epoch {epoch}, D Loss: {loss_d.numpy():.4f}, G Loss: {loss_g.numpy():.4f}")
```

## 模块化GAN损失函数

![](img/591e45cad1f1d4045687cb5917dafade_189.png)

为了使代码结构更清晰，我们可以将GAN的对抗训练过程封装成一个模块化的损失函数类。这样，训练循环看起来就更像标准的监督学习。

![](img/591e45cad1f1d4045687cb5917dafade_196.png)

![](img/591e45cad1f1d4045687cb5917dafade_198.png)

![](img/591e45cad1f1d4045687cb5917dafade_200.png)

```python
class GANLoss(ndl.nn.Module):
    def __init__(self, discriminator, d_optimizer):
        super().__init__()
        self.discriminator = discriminator
        self.d_optimizer = d_optimizer
        self.loss_fn = ndl.nn.SoftmaxLoss()

    def forward(self, x_fake, x_real):
        batch_size = x_fake.shape[0]
        # --- 判别器更新（隐藏在forward中）---
        # 假数据预测
        y_fake_pred = self.discriminator(x_fake.detach())
        # 真数据预测
        y_real_pred = self.discriminator(x_real)
        # 创建标签
        zeros = ndl.Tensor(np.zeros((batch_size,)))
        ones = ndl.Tensor(np.ones((batch_size,)))
        # 计算判别器损失
        loss_d = self.loss_fn(y_fake_pred, zeros) + self.loss_fn(y_real_pred, ones)
        # 更新判别器参数
        self.d_optimizer.zero_grad()
        loss_d.backward()
        self.d_optimizer.step()

        # --- 返回生成器损失 ---
        # 再次用更新后的判别器评估假数据
        y_fake_pred_new = self.discriminator(x_fake)
        # 生成器希望假数据被判断为真（标签1）
        loss_g = self.loss_fn(y_fake_pred_new, ones)
        return loss_g
```

```python
def train_gan_modular(data, model_G, batch_size=32, epochs=2000):
    model_D = Discriminator()
    opt_D = ndl.optim.SGD(model_D.parameters(), lr=0.01)
    opt_G = ndl.optim.SGD(model_G.parameters(), lr=0.01)
    gan_loss = GANLoss(model_D, opt_D)
    n = data.shape[0]
    for epoch in range(epochs):
        idx = np.random.randint(0, n, batch_size)
        x_real_batch = data[idx]
        z_batch = ndl.Tensor(np.random.randn(batch_size, 2))
        x_fake_batch = model_G(z_batch)
        # 前向传播计算生成器损失（内部已更新判别器）
        loss_g = gan_loss(x_fake_batch, x_real_batch)
        # 更新生成器
        opt_G.zero_grad()
        loss_g.backward()
        opt_G.step()
```
